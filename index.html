function doCalc() { const PV = parseFloat(document.getElementById('amt').value) || 0; const clientRateRaw = parseFloat(document.getElementById('c_rate').value) || 0; const realRateRaw = parseFloat(document.getElementById('r_rate').value) || 0; const clientRate = clientRateRaw / 100 / 12; const realRate = realRateRaw / 100 / 12; const n = parseInt(document.getElementById('period').value) || 0; if (PV <= 0 || n <= 0) { alert("請輸入貸款金額與期數"); return; } // 1. 計算月付額 let monthlyPay = (clientRate === 0) ? Math.ceil(PV / n) : Math.round(PV * (Math.pow(1 + clientRate, n) * clientRate) / (Math.pow(1 + clientRate, n) - 1)); // 2. 計算現值與補貼息 let subsidy = 0; let actualAmt = PV; // 關鍵修正：如果兩個利率相同，補貼息就是 0 if (clientRateRaw === realRateRaw) { subsidy = 0; actualAmt = PV; } else if (realRate > 0) { const pvOfAnnuity = monthlyPay * ((1 - Math.pow(1 + realRate, -n)) / realRate); subsidy = Math.round(PV - pvOfAnnuity); // 確保補貼息不為負數 if (subsidy < 0) subsidy = 0; actualAmt = PV - subsidy; } document.getElementById('res').style.display = 'block'; document.getElementById('m_pay').innerText = '$ ' + monthlyPay.toLocaleString(); document.getElementById('subsidy').innerText = '$ ' + subsidy.toLocaleString(); document.getElementById('actual_amt').innerText = '$ ' + actualAmt.toLocaleString(); }
